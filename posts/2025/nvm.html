# NVM

# NVM (Node Version Manager) 명령어 정리

# 0. NVM 설치

관리자 powershell

choco install nvm
1

# 1. NVM 설치 확인

nvm --version
1

# 2. 설치 가능한 Node.js 버전 목록 보기

nvm ls-remote
1

# 3. 특정 버전의 Node.js 설치

nvm install <버전>
1

예:

nvm install 18.17.1
1

# 4. 설치된 Node.js 버전 목록 보기

nvm ls
1

# 5. 특정 버전으로 전환

nvm use <버전>
1

예:

nvm use 16.20.2
nvm use 18.18.2
1
2

# 6. 기본 Node.js 버전 설정 (새로운 터미널에서도 유지됨)

nvm alias default <버전>
1

예:

nvm alias default 18.17.1
1

# 7. 현재 사용 중인 Node.js 버전 확인

nvm current
1

# 8. Node.js 최신 LTS 버전 설치 및 사용

nvm install --lts
nvm use --lts
1
2

# 9. Node.js 버전 삭제

nvm uninstall <버전>
1

예:

nvm uninstall 16.20.0
1

# 10. nvm 사용 일시 정지 (Node.js 시스템 버전 사용)

nvm deactivate
1
